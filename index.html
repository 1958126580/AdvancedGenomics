Quick Start

Get up and running in minutes with our simple installation and intuitive API

üì¶ Installation

Install directly from GitHub:

using Pkg
Pkg.add(url="https://github.com/1958126580/AdvancedGenomics")
using AdvancedGenomics

# Load data
G = read_genotypes("data.vcf")
y = read_phenotypes("traits.csv").trait

# Run GWAS pipeline with QC, PCA, and FDR correction
results = run_gwas_pipeline(G, y)

# Generate professional HTML report
generate_html_report(results, "gwas_report.html")

# Create visualizations
manhattan_plot(results.p_values)
qq_plot(results.p_values)
using AdvancedGenomics

# Load genotype and phenotype data
G = read_genotypes("genotypes.bed")
y = read_phenotypes("phenotypes.txt").trait

# Run genomic selection with 5-fold cross-validation
results = run_gs_pipeline(G, y, method=:gblup, n_folds=5)

# Generate comprehensive HTML report
generate_gs_html_report(results, "gs_report.html")

# Display results
println("Mean accuracy: ", results.mean_accuracy)
using AdvancedGenomics

# Load data
G = read_genotypes("data.vcf")
y = read_phenotypes("traits.csv").trait

# Build and train genomic transformer
model = GenomicTransformer(
    n_markers=size(G, 2),
    d_model=128,
    n_heads=8,
    n_layers=6
)

# Train with GPU acceleration
trained_model = train_model(model, G, y, epochs=100)

# Predict and evaluate
predictions = predict(trained_model, G_test)
